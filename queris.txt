    Estado INTEGER DEFAULT 1,
    Familia INTEGER,
    Costo_Promedio NUMERIC,

    Tipo INTEGER,
    Precio_Venta NUMERIC,
    ID_Empresa INTEGER,
    Fecha_Creacion DATE DEFAULT CURRENT_DATE,
    Usuario_Creador INTEGER,

    FOREIGN KEY (Familia) REFERENCES Familia(ID_Familia),
